package com.example.prgm3;


import android.content.Context;
import android.graphics.Canvas;
import android.graphics.Color;
import android.graphics.Paint;
import android.os.Bundle;
import android.view.View;


import androidx.appcompat.app.AppCompatActivity;


public class MainActivity extends AppCompatActivity {


   @Override
   protected void onCreate(Bundle savedInstanceState) {
       super.onCreate(savedInstanceState);


       // Create an anonymous View subclass with custom drawing
       View drawingView = new View(this) {
           Paint paint = new Paint();  // Paint object used for drawing
           Paint paint1 = new Paint();
           @Override
           protected void onDraw(Canvas canvas) {
               super.onDraw(canvas);


               paint.setColor(Color.RED);       // Set paint color to green
               paint.setStyle(Paint.Style.FILL); // Fill shapes, no stroke
               paint1.setColor(Color.BLUE);
               paint1.setStyle(Paint.Style.FILL);


               // Draw a filled horizontal line (a thick rectangle instead)
               // Since a line with FILL won't show thickness, draw a thin rect instead
               canvas.drawRect(50, 45, 500, 55, paint);


               // Draw a filled rectangle bounded by top-left (50,100) and bottom-right (500,300)
               canvas.drawRect(50, 100, 500, 300, paint);


               // Draw a filled circle centered at (275, 500) with radius 150
               canvas.drawCircle(275, 500, 150, paint1);
           }
       };


       // Set this view as the activity's content view to display it
       setContentView(drawingView);
   }
}
